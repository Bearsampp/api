76function FreeImage_GetVersion()
77{
78 global $FI;
79 static $pfn = null;
80
81 if($pfn === null)
82 $pfn = wb_get_function_address("FreeImage_GetVersion", $FI);
83
84 // Must use wb_peek because this function returns a string pointer
85
86 $version = wb_peek(wb_call_function($pfn));

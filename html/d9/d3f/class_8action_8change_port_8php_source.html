126 break;
127 }
128 $changePort = $this->bin->changePort( $port, true, $this->wbProgressBar );
129 if ( $changePort === true ) {
130 $this->bin->getService()->restart();

134 $boxTitle
135 );
136 $bearsamppWinbinder->destroyWindow( $window );
137 }
